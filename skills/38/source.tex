\begin{skill}
\begin{narrow}
\textcolor{blue}{Conditional Probability Notation}

Understanding what $P\left( A\,\vert\, B\right),$

$P\left( B\,\vert\, A\right)$ and $P\left( A\cap B \right)$ mean

\end{narrow}
%

\reason

%text
The notation used to express conditional 
probability has a very special -- and a very
precise -- meaning\newline 

The table below should help you remember
\underline{when to use which notation}

\begin{center}
\begin{tabular}{Ncc}
\midrule
\text{Notation} & \text{What you know} & \text{Probability to find} \\
\midrule 
P\left( A\,\vert\, B\right) & \text{B has happened} & \text{$A$ happened?} \\
\midrule 
P\left( B\,\vert\, A\right) & \text{A has happened} & \text{$B$ happened?} \\
\midrule 
P\left(A\cap B\right) & \text{Nothing} & \text{$A$ and $B$}  \\
& & \text{happening together} \\
\midrule 
\end{tabular}
\end{center} 

\end{skill}
